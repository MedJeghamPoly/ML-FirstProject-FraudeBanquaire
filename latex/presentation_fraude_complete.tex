\section{Méthodologie}

\begin{frame}{Chaîne de traitement (vue d'ensemble)}
  \begin{enumerate}
    \item EDA : distributions, corrélations, valeurs manquantes.
    \item Tri par \texttt{Time} ; \textbf{validation temporelle} (train = début, test = fin) --- plus réaliste qu'un split aléatoire.
    \item \textbf{Features} : \texttt{log1p(Amount)}, ratios vs moyenne mobile \emph{causale} (sans fuite).
    \item \textbf{Sélection} : L1 (logistique) sur échantillon train.
    \item Standardisation + \textbf{SMOTE} sur le train uniquement (\texttt{imblearn}).
    \item Modèles : LR, arbre, RF (GridSearch), XGBoost, LightGBM (optionnel).
    \item \textbf{Seuil} : PR + coût $C_{\mathrm{FN}}\cdot \mathrm{FN} + C_{\mathrm{FP}}\cdot \mathrm{FP}$.
    \item Calibration, \textbf{dérive} (KS, PSI), \textbf{SHAP} (modèle arborescent).
  \end{enumerate}
\end{frame}
